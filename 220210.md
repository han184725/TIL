#gravity
T = int(input())
for tc in range(1, T+1):
    n=int(input())
    lst=list(map(int, input().split()))
    max_drop = 0
    for i in range(n-1):
        now = lst[i]
        temp_drop = 0
        for j in range(i+1, n):
            if now > lst[j]:
                temp_drop += 1
        if temp_drop > max_drop:
            max_drop = temp_drop
print(f'#{tc} {max_drop}')

#babygin
T = int(input())
def check_babygin(numbers):
    counter = [0]*10
    is_babygin = 0
    for number in numbers:
        counter[number] += 1
    idx = 0
    while idx < len(counter):
        if counter[idx] >= 3:
            counter[idx] -= 3
            is_babygin += 1
            continue
        if idx < len(counter)-2:
            if counter[idx] and counter[idx+1] and counter[idx+2]:
                counter[idx] -= 1
                counter[idx+1] -= 1
                counter[idx+2] -= 1
                is_babygin += 1
                continue
        
        if is_babygin==2:
            return 1
        else:
            return 0
        idx += 1
        
for tc in range(1, T+1):
    numbers = list(map(int, input()))
    result = check_babygin(numbers)
    print(f'#{tc} {result}')
