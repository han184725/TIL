import sys

sys.stdin = open('input.txt')

T = int(input())

def dfs(cnt, x, y, num):
    global ans
    if cnt==7:
        ans.add(num)
        return
    for dx, dy in move:
        nx=x+dx
        ny=y+dy
        if 0<=nx<4 and 0<=ny<4:
            next=num+arr[nx][ny]
            if cnt==6 and next in ans:
                continue
            dfs(cnt+1, nx, ny, next)

for tc in range(1, T + 1):
    arr=[list(input().split()) for _ in range(4)]
    ans=set()
    move=((0,1),(0,-1),(1,0),(-1,0))
    for i in range(4):
        for j in range(4):
            dfs(1,i,j,arr[i][j])
    print(f'#{tc} {len(ans)}')
